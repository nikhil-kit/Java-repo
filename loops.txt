package jav1;

public class loops {
	public static void main(String args[]) {
		for(int i=0;i<101;i++)
			System.out.println(i);
		int m=100;
		System.out.println(m);
		m=m/5;
		System.out.println(m);
		m=m*5;
		System.out.println(m);
		m=m-5;
		System.out.println(m);
		for(int i=100;i>=1;i--)
			System.out.println(i);
		int f=1;
		while(f<=100)
			
			System.out.println(f);
	f+=1;
	int r=1;
	do {
		System.out.println(r);
		r+=1;
	}while(r<=10);
	}
	}
